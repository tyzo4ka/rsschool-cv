class ProductDetailView(DetailView):
    template_name = "products/detail.html"
    context_object_name = "product"
    model = Product

    def get_context_data(self, **kwargs):
        context = super(ProductDetailView, self).get_context_data(**kwargs)
        units = Unit.objects.filter(product_id=self.object.pk, quantity__gte=1).order_by('-size')
        logger.debug(self.object.category)

        if self.object.category.title == 'perfumery':
            context['sizes'] = {'title': 'Oбъем', 'values': units}
        else:
            context['sizes'] = {'title': 'Размер', 'values': units}
        return context
